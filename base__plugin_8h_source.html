202 /* check whether we get sensor data */
203 void healthCheck(const ros::TimerEvent & e)
204 {
205 boost::lock_guard<boost::mutex> lock(health_check_mutex_);
206
207 /* this will call only once, no recovery */
208 for(int i = 0; i < health_.size(); i++)
209 {
210 if(ros::Time::now().toSec() - health_stamp_.at(i) > health_timeout_ && health_.at(i)) // && !simulation_
211 {
212 ROS_ERROR("[%s, chan%d]: can not get fresh sensor data for %f[sec]", indexed_nhp_.getNamespace().c_str(), i, ros::Time::now().toSec() - health_stamp_.at(i));
213 /* TODO: the solution to unhealth should be more clever */
214 estimator_->setUnhealthLevel(unhealth_level_);
215
216 health_.at(i) = false;
217 }
218 }

255 inline const tf::Transform& getBaseLink2SensorTransform() const { return sensor_tf_; }
256
257 bool updateBaseLink2SensorTransform()
258 {
259 /* get transform from baselink to sensor frame */
260 if(!variable_sensor_tf_flag_ && get_sensor_tf_) return true;
261
262 /*
263 for joint or servo system, this should be processed every time,
264 therefore kinematics based on kinematics is better, since the tf need 0.x[sec].
265 */
266 const auto segments_tf = robot_model_->getSegmentsTf();
267
268 if(segments_tf.empty())
269 {
270 if(get_sensor_tf_) ROS_ERROR("the segment tf is empty after init phase");
271
272 ROS_DEBUG_STREAM("segment tf is empty");
273 return false;
274 }
275
276 if(segments_tf.find(sensor_frame_) == segments_tf.end())
277 {
278 ROS_ERROR_THROTTLE(0.5, "can not find %s in kinematics model", sensor_frame_.c_str());
279 return false;
280 }
281
282 try
283 {
284 tf::transformKDLToTF(segments_tf.at(robot_model_->getBaselinkName()).Inverse() * segments_tf.at(sensor_frame_), sensor_tf_);
285 }
286 catch (...)
287 {
288 ROS_ERROR("Bug: can not find %s in spite of segments_tf.find is true", sensor_frame_.c_str());
289 }
290
291 double y, p, r; sensor_tf_.getBasis().getRPY(r, p, y);
292 if(!variable_sensor_tf_flag_)
293 ROS_INFO("%s: get tf from %s to %s, [%f, %f, %f], [%f, %f, %f]",
294 indexed_nhp_.getNamespace().c_str(),
295 robot_model_->getBaselinkName().c_str(), sensor_frame_.c_str(),
296 sensor_tf_.getOrigin().x(), sensor_tf_.getOrigin().y(),
297 sensor_tf_.getOrigin().z(), r, p, y);
298
299 get_sensor_tf_ = true;
300 return true;
301 }
