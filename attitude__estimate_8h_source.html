128 void update()
129 {
130 if(gps_)
131 {
132 if(gps_->getMagValid() && !estimator_->getMagDecValid())
133 {
134 /* update magnetic declination by GPS receive data */
135 estimator_->setMagDeclination(gps_->getMagDeclination());
136 }
137 }
138
139
140 if(imu_list_[0]->getUpdate())
141 {
142 /* attitude estimation */
143 Vector3f gyro, acc, mag;
144 multiImuFusion(gyro, acc, mag);
145 estimator_->update(gyro, acc, mag);
146
147 /* send message to ros*/
148 if(nh_->connected()) publish();
149
150 /* reset update status of imu*/
151 imu_list_[0]->setUpdate(false);
152 }
153 }

155 void multiImuFusion(Vector3f& gyro, Vector3f& acc, Vector3f& mag )
156 {
157 /* only gyro method */
158 Vector3f sum_gyro = Vector3f();
159 for(unsigned int i = 0; i < imu_list_.size(); i++)
160 {
161 sum_gyro += (imu_list_[i]->getGyro() * imu_weights_[i]);
162 }
163 gyro = sum_gyro;
164 acc = imu_list_[0]->getAcc();
165 mag = imu_list_[0]->getMag();
166 }

180 /* send message via ros protocol */
181 void publish()
182 {
183 /* imu data (default: body/board frame */
184 uint32_t now_time = HAL_GetTick();
185 if( now_time - last_imu_pub_time_ >= IMU_PUB_INTERVAL)
186 {
187 last_imu_pub_time_ = now_time;
188#ifdef SIMULATION
189 imu_msg_.stamp = ros::Time::now();
190#else
191 imu_msg_.stamp = nh_->now();
192#endif
193 for(int i = 0; i < 3 ; i ++)
194 {
195 imu_msg_.mag_data[i] = estimator_->getMag(Frame::BODY)[i];
196 imu_msg_.acc_data[i] = estimator_->getAcc(Frame::BODY)[i];
197 imu_msg_.gyro_data[i] = estimator_->getAngular(Frame::BODY)[i];
198#if ESTIMATE_TYPE == COMPLEMENTARY
199 imu_msg_.angles[i] = estimator_->getAttitude(Frame::BODY)[i]; // get the attitude at body frame3
200#endif
201 }
202#ifdef SIMULATION
203 imu_pub_.publish(imu_msg_);
204#else
205 imu_pub_.publish(&imu_msg_);
206#endif
207 }
208
209 /* attitude data (default: cog/virtual frame) */
210 if( now_time - last_attitude_pub_time_ >= ATTITUDE_PUB_INTERVAL)
211 {
212 last_attitude_pub_time_ = now_time;
213#ifdef SIMULATION
214 attitude_msg_.header.stamp = ros::Time::now();
215#else
216 attitude_msg_.header.stamp = nh_->now();
217#endif
218
219#if ESTIMATE_TYPE == COMPLEMENTARY
220 /* get the attitude at virtual frame */
221 attitude_msg_.vector.x = estimator_->getAttitude(Frame::VIRTUAL).x;
222 attitude_msg_.vector.y = estimator_->getAttitude(Frame::VIRTUAL).y;
223 attitude_msg_.vector.z = estimator_->getAttitude(Frame::VIRTUAL).z;
224#endif
225#ifdef SIMULATION
226 attitude_pub_.publish(attitude_msg_);
227#else
228 attitude_pub_.publish(&attitude_msg_);
229#endif
