17#define GYR_CMPF_FACTOR 600
18#define GYR_CMPFM_FACTOR 250
19#define INV_GYR_CMPF_FACTOR (1.0f / (GYR_CMPF_FACTOR + 1.0f))
20#define INV_GYR_CMPFM_FACTOR (1.0f / (GYR_CMPFM_FACTOR + 1.0f))
21#define PRESCLAER_ACC 3 // if value=1, it means same rate with gyro, for genral attitude estimation
22
23#define G_MIN 72
24#define G_MAX 133
25
26class ComplementaryAHRS: public EstimatorAlgorithm
27{
28public:
29 ComplementaryAHRS():EstimatorAlgorithm(), est_g_(), est_m_(){}
30
31private:
32 std::array<ap::Vector3f, 2> est_g_, est_m_;
33
34 /* core esitmation process, using body frame */
35 void estimation()
36 {
37 EstimatorAlgorithm::estimation();
38
39 int valid_acc = 0;
40 static int cnt = 0;
41
42 float acc_magnitude = acc_[Frame::BODY] * acc_[Frame::BODY]; //norm?
43 ap::Vector3f est_g_b_tmp = est_g_[Frame::BODY];
44 ap::Vector3f est_m_b_tmp = est_m_[Frame::BODY];
45
46 ap::Vector3f gyro_rotate = gyro_[Frame::BODY] * DELTA_T;
47
48 est_m_[Frame::BODY] += (est_m_b_tmp % gyro_rotate ); //rotation by gyro
49 est_g_[Frame::BODY] += (est_g_b_tmp % gyro_rotate ); //rotation by gyro
50
51 if( G_MIN < acc_magnitude && acc_magnitude < G_MAX) valid_acc = 1;
52 else valid_acc = 0;
53
54 //*********************************************************************
55 //** Apply complimentary filter (Gyro drift correction)
56 //** If accel magnitude >1.15G or <0.85G and ACC vector outside of the limit range
57 //** => we neutralize the effect of accelerometers in the angle estimation.
58 //** To do that, we just skip filter, as EstV already rotated by Gyro
59 //*********************************************************************
60
61 est_g_b_tmp = est_g_[Frame::BODY];
62 est_m_b_tmp = est_m_[Frame::BODY];
63
64 /* acc correction */
65 if ( valid_acc == 1 && cnt == 0)
66 est_g_[Frame::BODY] = (est_g_b_tmp * GYR_CMPF_FACTOR + acc_[Frame::BODY]) * INV_GYR_CMPF_FACTOR;
67
68 /* mag correction */
69 if ( prev_mag_ != mag_[Frame::BODY] )
70 {
71 prev_mag_ = mag_[Frame::BODY];
72 est_m_[Frame::BODY] = (est_m_b_tmp * GYR_CMPFM_FACTOR + mag_[Frame::BODY]) * INV_GYR_CMPFM_FACTOR;
73 }
74
75 // Attitude of the estimated vector
76 float sq_g_x_sq_g_z = est_g_[Frame::BODY].x * est_g_[Frame::BODY].x + est_g_[Frame::BODY].z * est_g_[Frame::BODY].z;
77 float sq_g_y_sq_g_z = est_g_[Frame::BODY].y * est_g_[Frame::BODY].y + est_g_[Frame::BODY].z * est_g_[Frame::BODY].z;
78 float invG = ap::inv_sqrt(sq_g_x_sq_g_z + est_g_[Frame::BODY].y * est_g_[Frame::BODY].y);
79 rpy_[Frame::BODY].x = atan2f(est_g_[Frame::BODY].y , est_g_[Frame::BODY].z);
80 rpy_[Frame::BODY].y = atan2f(-est_g_[Frame::BODY].x , ap::inv_sqrt(sq_g_y_sq_g_z)* sq_g_y_sq_g_z);
81 rpy_[Frame::BODY].z = atan2f( est_m_[Frame::BODY].z * est_g_[Frame::BODY].y - est_m_[Frame::BODY].y * est_g_[Frame::BODY].z,
82 est_m_[Frame::BODY].x * invG * sq_g_y_sq_g_z - (est_m_[Frame::BODY].y * est_g_[Frame::BODY].y + est_m_[Frame::BODY].z * est_g_[Frame::BODY].z) * invG * est_g_[Frame::BODY].x ) + mag_declination_;
83
84
85 /* virtual(CoG) frame */
86 est_g_[Frame::VIRTUAL] = r_ * est_g_[Frame::BODY];
87 est_m_[Frame::VIRTUAL] = r_ * est_m_[Frame::BODY];
88 sq_g_x_sq_g_z = est_g_[Frame::VIRTUAL].x * est_g_[Frame::VIRTUAL].x + est_g_[Frame::VIRTUAL].z * est_g_[Frame::VIRTUAL].z;
89 sq_g_y_sq_g_z = est_g_[Frame::VIRTUAL].y * est_g_[Frame::VIRTUAL].y + est_g_[Frame::VIRTUAL].z * est_g_[Frame::VIRTUAL].z;
90 invG = ap::inv_sqrt(sq_g_x_sq_g_z + est_g_[Frame::VIRTUAL].y * est_g_[Frame::VIRTUAL].y);
91 rpy_[Frame::VIRTUAL].x = atan2f(est_g_[Frame::VIRTUAL].y , est_g_[Frame::VIRTUAL].z);
92 rpy_[Frame::VIRTUAL].y = atan2f(-est_g_[Frame::VIRTUAL].x , ap::inv_sqrt(sq_g_y_sq_g_z)* sq_g_y_sq_g_z);
93 rpy_[Frame::VIRTUAL].z = atan2f( est_m_[Frame::VIRTUAL].z * est_g_[Frame::VIRTUAL].y - est_m_[Frame::VIRTUAL].y * est_g_[Frame::VIRTUAL].z,
94 est_m_[Frame::VIRTUAL].x * invG * sq_g_y_sq_g_z - (est_m_[Frame::VIRTUAL].y * est_g_[Frame::VIRTUAL].y + est_m_[Frame::VIRTUAL].z * est_g_[Frame::VIRTUAL].z) * invG * est_g_[Frame::VIRTUAL].x ) + mag_declination_;
95 //********************************************************************************:
96 //** refrence1: https://sites.google.com/site/myimuestimationexperience/sensors/magnetometer
97 //** refrence2: http://uav.xenocross.net/hdg.html
98 //********************************************************************************
99
100 /* update */
101 if(valid_acc) cnt++;
102 if(cnt == PRESCLAER_ACC) cnt = 0;
103 }
104};
