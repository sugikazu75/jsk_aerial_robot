92 inline void setCurrAngleVal(const double& val, int value_type)
93 {
94 if(value_type == ValueType::BIT)
95 curr_angle_val_ = angle_scale_ * angle_sgn_ * (val - zero_point_offset_);
96 else if(value_type == ValueType::RADIAN)
97 curr_angle_val_ = val;
98 else
99 ROS_ERROR("wrong value type");
100
101 /* do low pass filtering */
102 if(filter_flag_)
103 curr_angle_val_ = lpf_angle_.filterFunction(curr_angle_val_);
104
105 if(!init_target_angle_val_)
106 {
107 target_angle_val_ = boost::algorithm::clamp(val, lower_limit_, upper_limit_);
108 init_target_angle_val_ = true;
109 }
110 }
111
112 inline void setTargetAngleVal(const double& val, int value_type)
113 {
114 if(value_type == ValueType::BIT)
115 target_angle_val_ = boost::algorithm::clamp(angle_scale_ * angle_sgn_ * (val - zero_point_offset_), lower_limit_, upper_limit_);
116 else if(value_type == ValueType::RADIAN)
117 target_angle_val_ = boost::algorithm::clamp(val, lower_limit_, upper_limit_);
118 else
119 ROS_ERROR("%s: wrong value type", name_.c_str());
120
121 if(!receive_real_state_) curr_angle_val_ = target_angle_val_;
122 }
123
124 inline void setCurrTorqueVal(const double& val)
125 {
126 curr_torque_val_ = torque_scale_ * angle_sgn_ * val;
127 }

153 }
154 else // ValueType::RADIAN

156 return target_angle_val_;
157 }
158 }
