93 template<typename T>
94 T saturate(const T val, const T min_val, const T max_val)
95 {
96 return std::min(std::max(val, min_val), max_val);
97 }

110
114 void enforceLimits(const ros::Duration& /* period */)
115 {
116 if(jh_.getForce() == 0) return;
117
118 /* because of "inline double setForce(double force) {*force_ = force;}", we can change the value with same address */
119 jh_.setForce(internal::saturate(jh_.getForce(), min_force_, max_force_));
120 }

122 private:
123 hardware_interface::RotorHandle jh_;

139 template <class HandleType>
140 class RotorLimitsInterface : public hardware_interface::ResourceManager<HandleType>
141 {
142 public:
143 HandleType getHandle(const std::string& name)
144 {
145 // Rethrow exception with a meaningful type
146 try
147 {
148 return this->hardware_interface::ResourceManager<HandleType>::getHandle(name);
149 }
150 catch(const std::logic_error& e)
151 {
152 throw joint_limits_interface::JointLimitsInterfaceException(e.what());
153 }
154 }
155
156 void enforceLimits(const ros::Duration& period)
157 {
158 typedef typename hardware_interface::ResourceManager<HandleType>::ResourceMap::iterator ItratorType;
159 for (ItratorType it = this->resource_map_.begin(); it != this->resource_map_.end(); ++it)
160 {
161 it->second.enforceLimits(period);
162 }
163 }
164 };
