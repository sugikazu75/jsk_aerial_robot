49 PID(const std::string name = std::string(""),
50 const double p_gain = 0, const double i_gain = 0, const double d_gain = 0,
51 const double limit_sum = 1e6, const double limit_p = 1e6, const double limit_i = 1e6, const double limit_d = 1e6,
52 const double limit_err_p = 1e6, const double limit_err_i = 1e6, const double limit_err_d = 1e6):
53 name_(name), result_(0), err_p_(0), err_i_(0), err_i_prev_(0), err_d_(0),
54 target_p_(0), target_d_(0), val_p_(0), val_d_(0),
55 p_term_(0), i_term_(0), d_term_(0)
56 {
57 setGains(p_gain, i_gain, d_gain);
58 setLimits(limit_sum, limit_p, limit_i, limit_d, limit_err_p, limit_err_i, limit_err_d);

61 virtual ~PID() = default;
62
63 virtual void update(const double err_p, const double du, const double err_d, const double feedforward_term = 0)
64 {
65 err_p_ = clamp(err_p, -limit_err_p_, limit_err_p_);
66 err_i_prev_ = err_i_;
67 err_i_ = clamp(err_i_ + err_p_ * du, -limit_err_i_, limit_err_i_);
68 err_d_ = clamp(err_d, -limit_err_d_, limit_err_d_);
69
70 p_term_ = clamp(err_p_ * p_gain_, -limit_p_, limit_p_);
71 i_term_ = clamp(err_i_ * i_gain_, -limit_i_, limit_i_);
72 d_term_ = clamp(err_d_ * d_gain_, -limit_d_, limit_d_);
73
74 result_ = clamp(p_term_ + i_term_ + d_term_ + feedforward_term, -limit_sum_, limit_sum_);
75 }
76
77 const double result() const { return result_; }

83 result_ = 0;
